# 返回值类型

//正确  声明函数返回值类型为字符串,实际返回也是字符串
function returnValue(): String {
  return "hello";
}

//错误 声明函数返回值类型为字符串,但实际返回数值
function returnValue(): String {
  return 100;
}

# 参数返回值类型

//正确 同时声明函数返回值类型及参数类型
function sumValue(value1: number, value2: number): number {
  return value1 + value2;
}
console.log(sumValue(10, 3)); //输出数字13

//错误 没有声明函数返回值类型及参数类型
function sumValue(value1: number, value2) {
  return value1 + value2;
}
console.log(sumValue(10, "")); //输出字符串13,相当于字符串拼接

# 函数类型

//声明
let myFuc: (a: number, b: number) => number;

//错误  直接修改或者直接赋值其他函数(如上方的sayHello)会报错
myFuc = sayHello; // 报错
myFuc = "直接修改类型"; // 报错

//正确
function sumValue(value1: number, value2: number): number {
  return value1 + value2;
}
myFuc = sumValue;
myFuc(10, 5);
